Xavfsizlik jadvali faollashtirilmagan! Qurilmalar buzib kirishi haqidagi hisobotlarni ko'rish uchun, Supabase console SQL Editor oynasida quyidagi kodni ishga tushiring: 
-- security_alerts jadvalini yaratish
create table public.security_alerts (
    id uuid default gen_random_uuid() primary key,
    device_id text not null,
    employee_name text,
    alert_type text not null,
    details text,
    severity text default 'HIGH' not null,
    created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

alter table public.security_alerts enable row level security;

create policy "Allow public insert to security_alerts" on public.security_alerts
    for insert to anon with check (true);

create policy "Allow Admin full access to security_alerts" on public.security_alerts
    for all to authenticated using (true);
