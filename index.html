Xodimlar Faoliyati Monitoringi

Tamg'alangan loyihalar ustida ishlash va local fayl yozish jarayonlari

Xavfsizlik tizimi jadvallari faollashtirilmagan! Qurilmalar buzib kirishi haqidagi hisobotlarni ko'rish, kutilayotgan faollashtirishlar, qora ro'yxat va faoliyat monitoringini ishlatish uchun Supabase console SQL Editor oynasida quyidagi kodni ishga tushiring: 
-- Jadvallarni yaratish va sozlash
create table if not exists public.security_alerts (
    id uuid default gen_random_uuid() primary key,
    device_id text not null,
    employee_name text,
    pc_name text,
    pc_user text,
    alert_type text not null,
    details text,
    severity text default 'HIGH' not null,
    created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

create table if not exists public.blacklist (
    id uuid default gen_random_uuid() primary key,
    device_id text not null unique,
    reason text,
    created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

create table if not exists public.pending_activations (
    id uuid default gen_random_uuid() primary key,
    device_id text not null unique,
    employee_name text not null,
    expires_at timestamp with time zone not null,
    created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

create table if not exists public.activity_logs (
    id uuid default gen_random_uuid() primary key,
    device_id text not null,
    employee_name text,
    pc_name text,
    pc_user text,
    activity_type text not null,
    details text,
    created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

alter table public.security_alerts enable row level security;
alter table public.blacklist enable row level security;
alter table public.pending_activations enable row level security;
alter table public.activity_logs enable row level security;

create policy "Allow public insert to security_alerts" on public.security_alerts for insert to anon with check (true);
create policy "Allow Admin full access to security_alerts" on public.security_alerts for all to authenticated using (true);

create policy "Allow public read of blacklist" on public.blacklist for select using (true);
create policy "Allow Admin full access to blacklist" on public.blacklist for all to authenticated using (true);

create policy "Allow public insert to pending_activations" on public.pending_activations for insert to anon with check (true);
create policy "Allow public read of pending_activations" on public.pending_activations for select using (true);
create policy "Allow Admin full access to pending_activations" on public.pending_activations for all to authenticated using (true);

create policy "Allow public insert to activity_logs" on public.activity_logs for insert to anon with check (true);
create policy "Allow Admin full access to activity_logs" on public.activity_logs for all to authenticated using (true);
